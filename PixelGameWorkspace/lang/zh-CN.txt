SG_Decoration=装饰
SG_Appearance=表现
SG_Transform=变换
SG_Lights=光源
SG_FeatureNodes=功能
SG_Data=数据

# Sprite
SN_Sprite=精灵
ST_Sprite=单贴图(无光照)
ST_Sprite_Texture=展示贴图
ST_Sprite_Color=顶点色
ST_Sprite_Anchor=锚点

# LitSprite
SN_LitSprite=精灵(受光)
ST_LitSprite=单贴图(带光照计算)
ST_LitSprite_Albedo=反射率贴图
ST_LitSprite_Normal=法线贴图
ST_LitSprite_Specular=高光贴图
ST_LitSprite_LightingLayer=光照层(0 -> 3 依次从近到远)
ST_LitSprite_Color=顶点色
ST_LitSprite_Cutoff=Alpha裁剪(用于Alpha Test)
ST_LitSprite_Anchor=锚点

# PointLight
SN_PointLight=点光
ST_PointLight=点光源(球形放射且强度随距离衰减)
ST_PointLight_Icon=图标
ST_PointLight_Texture=光强贴图
ST_PointLight_LightingLayer=光照层(0 -> 3 依次从近到远)
ST_PointLight_Color=光源颜色
ST_PointLight_Radius=光源半径
ST_PointLight_Intensity=光照强度

# PlayerSpawn
SN_PlayerSpawn=玩家出生点
ST_PlayerSpawn=玩家的出生点，一般在切地图点的附近放置。
ST_PlayerSpawn_IsGameStart=是否是游戏刚开始时的出生点，如果配置有多个这类节点，游戏中只会取到第一个。

# Export Settings
ST_ExportSettings_DefaultLevelName=游戏刚开始时加载的关卡名称
ST_ExportSettings_GlobalUniforms=需要导出的全局Uniforms，必须在pipeline_hook中设置且类型必须是Float4。
ST_ExportSettings_ExceptLevels=不参与导出的关卡列表(输入名称即可)
